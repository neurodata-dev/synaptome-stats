lab1 <- L$labels$col
rawLab <- lab1[order(lab1)]
p1 <- order(lab1)
distRaw <- as.matrix(dist(dat))
plot(raster(distRaw[p1,p1]), col = viridis(255)[1:215], main = "Distance matrix sorted by label: PDP^T")


rank_dist <- apply(distRaw[p1,p1], 1, rank, ties.method = "average")-1
plot(raster(log(rank_dist+1)), col = rev(gray.colors(255)), main = "rank distances")


relia <- rdf(rank_dist, lab1[p1])
hist(relia, main = "Reliability (line at chance)")
chance <- 1/N^2 * sum(table(lab1) * (N - table(lab1)))
abline(v = chance, lwd = 2, col = 'red')


discrim <- mnr(relia, lab1[p1], remove_outliers = FALSE, output = TRUE)
